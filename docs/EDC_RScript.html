    xlim = c(-1, 1), ylim = c(-1, 1),
    xlab = paste0("CP", ax1, " (", pcts[ax1], "%)"),
    ylab = paste0("CP", ax2, " (", pcts[ax2], "%)"),
    main = paste0("Cercle des corrélations - Plan CP", ax1, "-CP", ax2),
    type = "n", asp = 1
  )
  angles <- seq(0, 2 * pi, length = 1000)
  points(cos(angles), sin(angles), type = "l", col = "blue", lty = 2)
  abline(h = 0, v = 0, lty = 2, col = "grey")
  arrows(0, 0, corvar[, ax1] * 0.95, corvar[, ax2] * 0.95, length = 0.1, angle = 15)
  text(corvar[, ax1], corvar[, ax2], labels = rownames(corvar), cex = 0.8, col = "darkblue")
}

vip_scores <- function(model, ncomp){
  SS  <- c(model$Yloadings[, 1:ncomp])^2 * colSums(model$scores[, 1:ncomp]^2)
  W   <- model$loading.weights[, 1:ncomp]
  p   <- nrow(W)
  VIP <- sqrt(p * rowSums(sweep(W^2, 2, SS, "*")) / sum(SS))
  VIP
}
