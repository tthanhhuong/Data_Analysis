7.2.7 Choix du nombre optimal de composantes par CV répétée

set.seed(123)

PRESS100CV <- matrix(NA, 100, ncomp_max)
R2_100CV   <- matrix(NA, 100, ncomp_max)

for(i in 1:100){
  idx_cv_test <- sample(1:nrow(data_appCR), size = floor(nrow(data_appCR) / 2))
  idx_cv_app  <- setdiff(1:nrow(data_appCR), idx_cv_test)

  mod_cv <- plsr(Calories ~ ., data = data_appCR[idx_cv_app, ], ncomp = ncomp_max)

  for(h in 1:ncomp_max){
    pred_cv <- as.vector(predict(mod_cv, newdata = data_appCR[idx_cv_test, ], ncomp = h))
    obs_cv  <- data_appCR$Calories[idx_cv_test]

    PRESS100CV[i, h] <- sum((obs_cv - pred_cv)^2)
    R2_100CV[i, h] <- 1 - sum((obs_cv - pred_cv)^2) /
      sum((obs_cv - mean(data_appCR$Calories[idx_cv_app]))^2)
  }
}

PRESS100moy <- colMeans(PRESS100CV, na.rm = TRUE)
R2_100moy   <- colMeans(R2_100CV, na.rm = TRUE)

par(mfrow = c(1, 2))

     xlab = "Nb de composantes", ylab = "PRESS moyen",
     main = "PRESS moyen (100 runs)")

7.2.10 Scores VIP

vip_scores <- function(model, ncomp){
  SS  <- c(model$Yloadings[, 1:ncomp])^2 * colSums(model$scores[, 1:ncomp]^2)
  W   <- model$loading.weights[, 1:ncomp]
  p   <- nrow(W)
  VIP <- sqrt(p * rowSums(sweep(W^2, 2, SS, "*")) / sum(SS))
  VIP
}

vip <- vip_scores(respls, ncomp = ncomp_opt)
